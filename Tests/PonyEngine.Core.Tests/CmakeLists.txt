message(STATUS "Configuring PonyEngine.Core.Tests")
add_executable(PonyEngine.Core.Tests)

message(STATUS "Configuring sources")
target_sources(PonyEngine.Core.Tests PRIVATE
	"Log/LogFormat.cpp"
	"Log/LogHelper.cpp"
	"Log/LogMacro.cpp"
	"Utility/Enum.cpp"
	"Utility/SafeFormat.cpp"
)

if(PONY_WINCORE)
	message(STATUS "Configuring sources for WinCore")
	target_sources(PonyEngine.Core.Tests PRIVATE
		"Platform/WinCore/GUID.cpp"
		"Platform/WinCore/Utility.cpp"
	)
endif()

if(PONY_WINDOWS)
	message(STATUS "Configuring sources for Windows")
	target_sources(PonyEngine.Core.Tests PRIVATE
		"Platform/Windows/Cursor.cpp"
		"Platform/Windows/Module.cpp"
	)
endif()

message(STATUS "Configuring defines")
pony_set_log_defines(PonyEngine.Core.Tests TRUE)

message(STATUS "Setting build options")
set_target_properties(PonyEngine.Core.Tests PROPERTIES CXX_STANDARD 23)
pony_set_build_options(PonyEngine.Core.Tests TRUE)

message(STATUS "Configuring dependencies")
target_link_libraries(PonyEngine.Core.Tests PRIVATE 
	Catch2::Catch2WithMain
	PonyEngine.Core
)

message(STATUS "Discovering tests")
include(CTest)
include(Catch)
catch_discover_tests(PonyEngine.Core.Tests)
