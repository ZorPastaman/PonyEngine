	"Math/Ball.cpp"
	"Math/BallInsides.cpp"
	"Math/BallIntersections.cpp"

	"Math/Box.cpp"
	"Math/BoxInsides.cpp"
	"Math/BoxIntersections.cpp"
	"Math/Color.cpp"
	"Math/Common.cpp"
	"Math/Flat.cpp"
	"Math/FlatIntersections.cpp"
	"Math/Matrix.cpp"
	"Math/OrientedBox.cpp"
	"Math/OrientedBoxInsides.cpp"

)

if(PONY_WINCORE)
	message(VERBOSE "Configuring sources for WinCore")
	target_sources(PonyEngine.Core.Tests PRIVATE
		"Platform/WinCore/GUID.cpp"
		"Platform/WinCore/Utility.cpp"
	)
endif()

if(PONY_WINDOWS)
	message(VERBOSE "Configuring sources for Windows")
	target_sources(PonyEngine.Core.Tests PRIVATE
		"Platform/Windows/Cursor.cpp"
		"Platform/Windows/Module.cpp"
	)
endif()

message(VERBOSE "Configuring defines")
pony_set_log_defines(PonyEngine.Core.Tests ${PONY_ENGINE_LOG_LEVEL} ${PONY_ENGINE_LOG_STACKTRACE_LEVEL} ${PONY_CONSOLE_LOG} ${PONY_PLATFORM_CONSOLE_LOG})
if(PONY_ENGINE_TESTING_BENCHMARK)
	target_compile_definitions(PonyEngine.Core.Tests PRIVATE PONY_ENGINE_TESTING_BENCHMARK)
endif()

message(VERBOSE "Setting properties")
set_target_properties(PonyEngine.Core.Tests PROPERTIES 
	CXX_STANDARD 23
	POSITION_INDEPENDENT_CODE TRUE
)

message(VERBOSE "Setting build options")
pony_set_build_options(PonyEngine.Core.Tests ${PONY_ENGINE_OPTIMIZATION})

message(VERBOSE "Configuring dependencies")
target_link_libraries(PonyEngine.Core.Tests PRIVATE 
	Catch2::Catch2WithMain
	PonyEngine.Core
)

message(VERBOSE "Discovering tests")
catch_discover_tests(PonyEngine.Core.Tests)
