	"File/Path.cpp"
	"ID/WinGUID.cpp"
	"Log/LogFormat.cpp"
	"Log/LogMacro.cpp"
	"Math/Ball.cpp"
	"Math/BallInsides.cpp"

	"Math/Bounds.cpp"
	"Math/Box.cpp"
	"Math/BoxInsides.cpp"

	"Math/Color.cpp"
	"Math/Common.cpp"
	"Math/Flat.cpp"

	"Math/Matrix.cpp"

	target_compile_definitions(PonyEngine.Core.Tests PRIVATE PONY_ENGINE_TESTING_BENCHMARK)
endif()

message(VERBOSE "Setting properties")
set_target_properties(PonyEngine.Core.Tests PROPERTIES 
	CXX_STANDARD 23
	POSITION_INDEPENDENT_CODE TRUE
)

pony_set_build_options(PonyEngine.Core.Tests ${PONY_ENGINE_OPTIMIZATION})

message(VERBOSE "Configuring dependencies")
target_link_libraries(PonyEngine.Core.Tests PRIVATE 
	Catch2::Catch2WithMain
	PonyEngine.Core
)

message(VERBOSE "Discovering tests")
catch_discover_tests(PonyEngine.Core.Tests)
