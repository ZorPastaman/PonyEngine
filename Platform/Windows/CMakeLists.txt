set(PONY_TARGET_WINDOWS_SDK "10.0.26100.0")

if(CMAKE_SYSTEM_VERSION VERSION_LESS PONY_TARGET_WINDOWS_SDK)
	message(FATAL_ERROR "Building with Windows SDK '${CMAKE_SYSTEM_VERSION}', but ${PONY_TARGET_WINDOWS_SDK} or newer is required")
endif()

		message(FATAL_ERROR "Building with Windows SDK '${CMAKE_SYSTEM_VERSION}', but ${PONY_TARGET_WINDOWS_SDK} is required for engine build")
	endif()
endif()
