set(PONY_TARGET_WINDOWS_SDK_VERSION "10.0.26100.0")
string(REGEX REPLACE "\\\\$" "" PONY_WINDOWS_SDK_VERSION "$ENV{WindowsSDKVersion}")

if(NOT PONY_WINDOWS_SDK_VERSION)
	message(FATAL_ERROR "Windows SDK not set")
endif()
if(PONY_WINDOWS_SDK_VERSION VERSION_LESS PONY_TARGET_WINDOWS_SDK_VERSION)
	message(FATAL_ERROR "Building with Windows SDK 'PONY_WINDOWS_SDK_VERSION', but ${PONY_TARGET_WINDOWS_SDK_VERSION} or newer is required")
endif()
if(NOT PONY_IS_SUBPROJECT)
	if(NOT PONY_WINDOWS_SDK_VERSION VERSION_EQUAL PONY_TARGET_WINDOWS_SDK_VERSION)
		message(FATAL_ERROR "Building with Windows SDK 'PONY_WINDOWS_SDK_VERSION', but ${PONY_TARGET_WINDOWS_SDK_VERSION} is required for engine build")
	endif()
endif()
