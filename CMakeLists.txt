cmake_minimum_required (VERSION 3.31)

message(STATUS "Configuring PonyEngine environment")
if(NOT PONY_BUILD_TYPE STREQUAL "Engine" AND NOT PONY_BUILD_TYPE STREQUAL "Game" AND NOT PONY_BUILD_TYPE STREQUAL "Custom")
	message(FATAL_ERROR "Incorrect PONY_BUILD_TYPE. Allowed values: Engine, Game and Custom")
endif()
add_subdirectory("CMake/Build")
set_property(GLOBAL PROPERTY PONY_ENGINE_PATH "${CMAKE_CURRENT_LIST_DIR}/Engine")
if(PONY_BUILD_TYPE STREQUAL "Engine" OR PONY_BUILD_TYPE STREQUAL "Game")
	set(CMAKE_CXX_STANDARD 23)
	set(CMAKE_EXPERIMENTAL_CXX_IMPORT_STD "0e5b6991-d74f-4b3d-a41c-cf096e0b2508")
	set(CMAKE_CXX_STANDARD_REQUIRED ON)
	set(CMAKE_CXX_MODULE_STD 1)
	set(CMAKE_EXPERIMENTAL_CXX_MODULE_DYNDEP 1)
	set(CMAKE_EXPERIMENTAL_CXX_MODULE_CMAKE_API 1)
endif()

if(PONY_BUILD_TYPE STREQUAL "Engine" OR PONY_BUILD_TYPE STREQUAL "Game")
	message(STATUS "Configuring project")
	project(${PONY_GAME_NAME} VERSION ${PONY_GAME_VERSION} LANGUAGES CXX)
	pony_clean_build_options()

	message(STATUS "Validating settings")
	if(PONY_WINDOWS)
		if(CMAKE_SYSTEM_VERSION VERSION_LESS "10.0.26100.0")
			message(FATAL_ERROR "Building with Windows SDK '${CMAKE_HOST_SYSTEM_VERSION}', but 10.0.26100.0 or newer is required")
		endif()
	else()
		message(FATAL_ERROR "Unsupported platform!")
	endif()
	if(PONY_WINCORE AND NOT PONY_WINDOWS)
		message(FATAL_ERROR "PONY_WINCORE is used but PONY_WINDOWS isn't")
	endif()

	message(STATUS "Initializing properties")
	pony_init_install()
endif()

if(PONY_BUILD_TYPE STREQUAL "Engine")
	get_property(PONY_ENGINE_PATH_VALUE GLOBAL PROPERTY PONY_ENGINE_PATH)

	message(STATUS "Configuring log module settings")
	set(PONY_ENGINE_LOG_CORE ON)
	set(PONY_ENGINE_LOG_CONSOLE ON)
	set(PONY_ENGINE_LOG_PLATFORM_CONSOLE ON)
	set(PONY_ENGINE_LOG_FILE ON)
	set(PONY_ENGINE_LOG_CONSOLE_ORDER 10)
	set(PONY_ENGINE_LOG_PLATFORM_CONSOLE_ORDER 20)
	set(PONY_ENGINE_LOG_FILE_ORDER 50)

	message(STATUS "Configuring modules")
	add_subdirectory("${PONY_ENGINE_PATH_VALUE}/PonyEngine.Core")
	add_subdirectory("${PONY_ENGINE_PATH_VALUE}/PonyEngine.Log.Extension")
	add_subdirectory("${PONY_ENGINE_PATH_VALUE}/PonyEngine.Log.Implementation")

	message(STATUS "Configuring application")
	add_subdirectory("${PONY_ENGINE_PATH_VALUE}/PonyEngine.Application")

	message(STATUS "Configuring test project")
	project("PonyEngineTests" LANGUAGES CXX)
	add_subdirectory("Tests")
endif()
