cmake_minimum_required (VERSION 3.31)
message(STATUS "Configuring PonyEngine")

if(PROJECT_NAME)
	message(VERBOSE "Configuring as sub-project")
	set(PONY_IS_SUBPROJECT TRUE)
else()
	message(VERBOSE "Configuring as main project")
	set(PONY_IS_SUBPROJECT FALSE)
endif()

if(NOT PONY_IS_SUBPROJECT)
	message(VERBOSE "Configuring environment")
	set(CMAKE_CXX_STANDARD 23)
	set(CMAKE_CXX_STANDARD_REQUIRED ON)
	set(CMAKE_EXPERIMENTAL_CXX_IMPORT_STD "0e5b6991-d74f-4b3d-a41c-cf096e0b2508")
	set(CMAKE_CXX_MODULE_STD ON)
endif()
include(FetchContent)

message(VERBOSE "Configuring project")
project("PonyEngine" VERSION "0.0.0.0" LANGUAGES CXX)
if(NOT PONY_IS_SUBPROJECT)
	pony_clean_build_options()
endif()

message(VERBOSE "Configuring platform: ${CMAKE_SYSTEM_NAME}")
if(PONY_WINCORE OR PONY_WINDOWS)
	message(FATAL_ERROR "Platform defines are preset")
endif()
if(CMAKE_SYSTEM_NAME STREQUAL "Windows")
	if(CMAKE_SYSTEM_VERSION VERSION_LESS "10.0.26100.0")
		message(FATAL_ERROR "Building with Windows SDK '${CMAKE_SYSTEM_VERSION}', but 10.0.26100.0 or newer is required")
	endif()
	if(NOT PONY_IS_SUBPROJECT)
		if(NOT CMAKE_SYSTEM_VERSION VERSION_EQUAL "10.0.26100.0")
			message(FATAL_ERROR "Building with Windows SDK '${CMAKE_SYSTEM_VERSION}', but 10.0.26100.0 is required for engine build")
		endif()
	endif()

	set(PONY_WINCORE "TRUE")
	set(PONY_WINDOWS "TRUE")

	if(PONY_IS_SUBPROJECT)
		set(PONY_WINCORE "TRUE" PARENT_SCOPE)
		set(PONY_WINDOWS "TRUE" PARENT_SCOPE)
	endif()
else()
	message(FATAL_ERROR "Unsupported platform!")
endif()

message(VERBOSE "Configuring modules")
set(PONY_INSTALL_TARGETS "")
add_subdirectory("Engine/PonyEngine.Core")
if(NOT PONY_IS_SUBPROJECT)
	message(VERBOSE "Post-configuring")
	pony_set_log_defines(PonyEngine.Core ${PONY_ENGINE_LOG_LEVEL} ${PONY_ENGINE_LOG_STACKTRACE_LEVEL} ${PONY_CONSOLE_LOG} ${PONY_PLATFORM_CONSOLE_LOG})
	pony_set_build_options(PonyEngine.Core ${PONY_ENGINE_OPTIMIZATION})
endif()

if(PONY_ENGINE_LOG_EXT OR PONY_ENGINE_LOG_IMPL)
	add_subdirectory("Engine/PonyEngine.Log.Ext")
	if(NOT PONY_IS_SUBPROJECT)
		message(VERBOSE "Post-configuring")
		pony_set_log_defines(PonyEngine.Log.Ext ${PONY_ENGINE_LOG_LEVEL} ${PONY_ENGINE_LOG_STACKTRACE_LEVEL} ${PONY_CONSOLE_LOG} ${PONY_PLATFORM_CONSOLE_LOG})
		pony_set_build_options(PonyEngine.Log.Ext ${PONY_ENGINE_OPTIMIZATION})
	endif()
endif()

if(PONY_ENGINE_LOG_IMPL)
	add_subdirectory("Engine/PonyEngine.Log.Impl")
	if(NOT PONY_IS_SUBPROJECT)
		message(VERBOSE "Post-configuring")
		pony_set_log_defines(PonyEngine.Log.Impl ${PONY_ENGINE_LOG_LEVEL} ${PONY_ENGINE_LOG_STACKTRACE_LEVEL} ${PONY_CONSOLE_LOG} ${PONY_PLATFORM_CONSOLE_LOG})
		pony_set_build_options(PonyEngine.Log.Impl ${PONY_ENGINE_OPTIMIZATION})

		if(NOT PONY_ENGINE_FORCE_STATIC)
			list(APPEND PONY_INSTALL_TARGETS PonyEngine.Log.Impl)
		endif()
	endif()
endif()

if(PONY_ENGINE_ENGINE_EXT OR PONY_ENGINE_ENGINE_IMPL)
	add_subdirectory("Engine/PonyEngine.Engine.Ext")
	if(NOT PONY_IS_SUBPROJECT)
		message(VERBOSE "Post-configuring")
		pony_set_log_defines(PonyEngine.Engine.Ext ${PONY_ENGINE_LOG_LEVEL} ${PONY_ENGINE_LOG_STACKTRACE_LEVEL} ${PONY_CONSOLE_LOG} ${PONY_PLATFORM_CONSOLE_LOG})
		pony_set_build_options(PonyEngine.Engine.Ext ${PONY_ENGINE_OPTIMIZATION})
	endif()
endif()

if(PONY_ENGINE_ENGINE_IMPL)
	add_subdirectory("Engine/PonyEngine.Engine.Impl")
	if(NOT PONY_IS_SUBPROJECT)
		message(VERBOSE "Post-configuring")
		pony_set_log_defines(PonyEngine.Engine.Impl ${PONY_ENGINE_LOG_LEVEL} ${PONY_ENGINE_LOG_STACKTRACE_LEVEL} ${PONY_CONSOLE_LOG} ${PONY_PLATFORM_CONSOLE_LOG})
		pony_set_build_options(PonyEngine.Engine.Impl ${PONY_ENGINE_OPTIMIZATION})

		if(NOT PONY_ENGINE_FORCE_STATIC)
			list(APPEND PONY_INSTALL_TARGETS PonyEngine.Engine.Impl)
		endif()
	endif()
endif()

if(PONY_ENGINE_APPLICATION)
	add_subdirectory("Engine/PonyEngine.Application")
	if(NOT PONY_IS_SUBPROJECT)
		message(VERBOSE "Post-configuring")
		pony_set_log_defines(PonyEngine.Application ${PONY_ENGINE_LOG_LEVEL} ${PONY_ENGINE_LOG_STACKTRACE_LEVEL} ${PONY_CONSOLE_LOG} ${PONY_PLATFORM_CONSOLE_LOG})
		pony_set_build_options(PonyEngine.Application ${PONY_ENGINE_OPTIMIZATION})

		list(APPEND PONY_INSTALL_TARGETS PonyEngine.Application)
	endif()

	message(VERBOSE "Adding modules")
	if(TARGET PonyEngine.Log.Impl)
		target_link_libraries(PonyEngine.Application PRIVATE PonyEngine.Log.Impl)
	endif()
	if(TARGET PonyEngine.Engine.Impl)
		target_link_libraries(PonyEngine.Application PRIVATE PonyEngine.Engine.Impl)
	endif()
endif()

if(NOT PONY_IS_SUBPROJECT AND PONY_ENGINE_ADD_GAME)
	message(VERBOSE "Adding game")
	add_subdirectory("Game/MinimalSample/GameSample")
	pony_set_log_defines(GameSample ${PONY_GAME_LOG_LEVEL} ${PONY_GAME_LOG_STACKTRACE_LEVEL} ${PONY_CONSOLE_LOG} ${PONY_PLATFORM_CONSOLE_LOG})
	pony_set_build_options(GameSample ${PONY_GAME_OPTIMIZATION})

	if(TARGET PonyEngine.Application)
		target_link_libraries(PonyEngine.Application PRIVATE GameSample)
	endif()

	if(NOT PONY_ENGINE_FORCE_STATIC)
		list(APPEND PONY_INSTALL_TARGETS GameSample)
	endif()
endif()

if(NOT PONY_IS_SUBPROJECT)
	message(VERBOSE "Installing")
	install(TARGETS ${PONY_INSTALL_TARGETS}
		RUNTIME DESTINATION .
		LIBRARY DESTINATION
	)
endif()

if(NOT PONY_IS_SUBPROJECT AND PONY_ENGINE_TESTING)
	message(STATUS "Configuring test project")
	add_subdirectory("Tests")
endif()
